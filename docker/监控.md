### influxdb安装
- 安装
~~~
docker run -d --name influxdb -p 8083:8086 --hostname=influxdb influxdb
~~~
- 创建db
~~~
# 进入influxdb镜像
docker exec -it influxdb influx
# 创建db
CREATE DATABASE "数据库名称"
# 创建账号密码
CREATE USER "root" WITH PASSWORD 'root' WITH ALL PRIVILEAGES
~~~
